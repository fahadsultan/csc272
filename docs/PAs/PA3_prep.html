for year in [2015, 2017]: 
    data = pd.read_csv('../data/US Census/acs%s_county_data.csv' % year)
    data = data.drop(columns=['IncomePerCap', 'IncomePerCapErr'])
    for state in states:
        state_data = data[data['State']==state]
        state_data = state_data.drop(columns=['State'])
        state_name = state.replace(' ', '')
        state_data.to_csv('../data/PA3/%s/%s.csv' % (year, state_name), index=False)
